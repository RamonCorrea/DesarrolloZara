CREATE PROCEDURE [dbo].PROC_WEB_UPD_PLANTURNO (@COD_INTERNO VARCHAR(11),
					    		@FECHA 		DATETIME,
							@COD_EMPRESA 	VARCHAR(4),
							@COD_TURNO 	VARCHAR(5))		
UPDATE PLAN_TURNOS SET
COD_TURNO = @COD_TURNO,
WHERE COD_INTERNO = @COD_INTERNO 
AND COD_EMPRESA=@COD_EMPRESA AND FECHA = @FECHA


CREATE PROCEDURE dbo.SP_WEB_INS_PLANTURNO (@COD_INTERNO 	VARCHAR(11),
						   @FECHA 		DATETIME,
						   @COD_EMPRESA 	VARCHAR(4),
						   @COD_TURNO 	VARCHAR(5))
AS		
INSERT INTO PLAN_TURNOS (COD_INTERNO,FECHA,COD_EMPRESA,COD_TURNO,SEMANA) VALUES
(@COD_INTERNO,@FECHA,@COD_EMPRESA,@COD_TURNO,@SEMANA)
