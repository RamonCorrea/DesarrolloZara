CREATE PROCEDURE [dbo].PROC_WEB_UPD_PLANTURNO (@COD_INTERNO VARCHAR(11),
					    		@FECHA 		DATETIME,
							@COD_EMPRESA 	VARCHAR(4),
							@COD_TURNO 	VARCHAR(5))		
UPDATE PLAN_TURNOS SET
COD_TURNO = @COD_TURNO,
WHERE COD_INTERNO = @COD_INTERNO 
AND COD_EMPRESA=@COD_EMPRESA AND FECHA = @FECHA


CREATE PROCEDURE dbo.SP_WEB_INS_PLANTURNO (@COD_INTERNO 	VARCHAR(11),
						   @FECHA 		DATETIME,
						   @COD_EMPRESA 	VARCHAR(4),
						   @COD_TURNO 	VARCHAR(5))
AS		
INSERT INTO PLAN_TURNOS (COD_INTERNO,FECHA,COD_EMPRESA,COD_TURNO,SEMANA) VALUES
(@COD_INTERNO,@FECHA,@COD_EMPRESA,@COD_TURNO,@SEMANA)


set ANSI_NULLS ON
set QUOTED_IDENTIFIER ON
GO
ALTER PROCEDURE [dbo].[PROC_TURNO_EMPLEADO](
				@COD_INTERNO VARCHAR(20),
				@FECHA_INI VARCHAR(13),
				@FECHA_FIN VARCHAR(13))

AS
BEGIN
	SELECT SUBSTRING(CONVERT(VARCHAR(12),FECHA,103),1,2) as FECHA, COD_TURNO FROM PLAN_TURNOS
	WHERE COD_INTERNO = @COD_INTERNO AND FECHA BETWEEN @FECHA_INI AND @FECHA_FIN
END


set ANSI_NULLS ON
set QUOTED_IDENTIFIER ON
GO

ALTER PROCEDURE [dbo].[SP_VERIFICA_NET] @USUARIO VARCHAR(20), @PASS VARCHAR(20) OUTPUT
AS
	SELECT @PASS = PWD_USUARIO FROM T_USUARIOS
	WHERE @USUARIO = ID_USUARIO
